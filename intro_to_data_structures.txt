	function addNewPerson (array, currentPerson, newPerson){
		let currentPersonIndex = array.indexOf(currentPerson);
		array.splice(currentPersonIndex, 0, newPerson);
		return array;
		}
	
	console.log(addNewPerson(array, 'Ava', 'Melissa'));
